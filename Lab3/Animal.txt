
public class Animal {
	    private String species;
	    int yearsOfLife;


	    public Animal (String species, int yearsOfLife) {
	        setSpecies(species);
	        setYearsOfLife(yearsOfLife);
	    }

	    @Override
	    public String toString() {
	        return " Animal called " + species  + " live for " + this.yearsOfLife + "\n";

	    }    

	    public int getYearsOfLife() {
			return yearsOfLife;
		}

		public void setYearsOfLife(int yearsOfLife) {
			this.yearsOfLife = yearsOfLife;
		}

		public String getSpecies() {
	        return species;
	    }

	    public void setSpecies(String species) {
	        this.species = species;
	    }
}
